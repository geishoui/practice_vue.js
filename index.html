リストの更新

index.html
    <input v-bind:value="myGoods.name">
    <input v-bind:value="myGoods.score">
    <button v-on:click="myAddGoods">項目を追加</button>
    <ul>
        <li v-for="item in myGoodsList" v-bind:key="item.id">
        ID: {{ item.id }}, name: {{ item.name }}, score: {{ item.score }}
            <button v-on:click="myRemoveGoods(index)">削除</button>
        </li>
    </ul>
main.js
    data: {
        myGoods: {
            name: '項目名',
            score: 289,            
        },
        myGoodsList: [
            {id: 1, name: 'hoge', score: 100 },
            {id: 2, name: 'fuga', score: 200 },
            {id: 3, name: 'piyo', score: 500 },
        ],
    },
    // アプリケーションで使用するメソッド
    methods: {
        myAddGoods: function() {
            // IDの最大値を取得
            var max = this.myGoodsList.reduce(function(a,b) {
                return a.id > b.id ? a.id : b.id
            }, 0)
            //push()で配列にデータを追加
            this.myGoodsList.push({
                id: max + 1,
                name: this.myGoods.name,
                score: this.myGoods.score,
            })
        },
        myRemoveGoods: function(index) {
            // spliceで受け取ったindexの位置から1個の要素を削除
            this.myGoodsList.splice(index, 1)
        },
    },

実際の動作

  • ID: {{item.id}}, name: {{item.name}}, score: {{item.score}}
